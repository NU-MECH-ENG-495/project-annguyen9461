216void ControlTurn::gradual_transition(int target_position) {
217 const int step_size = 60;
218 const int delay_ms = 15;
220 // Get current position
221 update_present_positions();
222 int current_position = present_positions[1]; // Using motor ID 1
223
224 // Calculate step increment for smooth transition
225 float step = static_cast<float>(target_position - current_position) / step_size;
226
227 // Execute the gradual transition through multiple smaller steps
228 for (int i = 1; i <= step_size; i++) {
229 int position = current_position + std::round(step * i);
230
231 // Split the position value into 4 bytes as required by Dynamixel protocol
232 uint8_t param_goal_position[4];
233 param_goal_position[0] = DXL_LOBYTE(DXL_LOWORD(position)); // Lower byte of lower word
234 param_goal_position[1] = DXL_HIBYTE(DXL_LOWORD(position)); // Higher byte of lower word
235 param_goal_position[2] = DXL_LOBYTE(DXL_HIWORD(position)); // Lower byte of higher word
236 param_goal_position[3] = DXL_HIBYTE(DXL_HIWORD(position)); // Higher byte of higher word
237
238 // Clear previous parameters and add new one
239 groupSyncWrite->clearParam();
240 if (!groupSyncWrite->addParam(1, param_goal_position)) { // Using motor ID 1
241 RCLCPP_WARN(this->get_logger(), "SyncWrite addParam failed");
242 continue;
243 }
244
245 // Send the packet to the motor
246 int dxl_comm_result = groupSyncWrite->txPacket();
247 if (dxl_comm_result != COMM_SUCCESS) {
248 RCLCPP_ERROR(this->get_logger(), "SyncWrite Failed: %s",
249 packetHandler->getTxRxResult(dxl_comm_result));
250 }
251
252 // Short pause between steps for smooth movement
253 std::this_thread::sleep_for(std::chrono::milliseconds(delay_ms));
254 }
255
256 // Final step to ensure exact target position is reached
257 uint8_t param_goal_position[4];
258 param_goal_position[0] = DXL_LOBYTE(DXL_LOWORD(target_position));
259 param_goal_position[1] = DXL_HIBYTE(DXL_LOWORD(target_position));
260 param_goal_position[2] = DXL_LOBYTE(DXL_HIWORD(target_position));
261 param_goal_position[3] = DXL_HIBYTE(DXL_HIWORD(target_position));
262
263 groupSyncWrite->clearParam();
264 groupSyncWrite->addParam(1, param_goal_position);
265 groupSyncWrite->txPacket();
266}
